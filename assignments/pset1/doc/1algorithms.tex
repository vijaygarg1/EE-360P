\begin{enumerate}
\item PSort

\begin{itemize}
\item Set pivot = begin
\item While begin < end
\begin{itemize}
\item Increment begin until A[begin] > A[pivot]
\item Decrement tempEnd until A[end] < A[pivot]
\item If begin < end, swap A[begin] and A[end]
\end{itemize}
\item Swap A[pivot] and A[end]
\item Split into two threads, one for each side of pivot's new location
\item Recursively call PSort for each thread
\end{itemize}

\item PSearch
\begin{itemize}
\item Divide size of array by number of threads
\item Create indices at division points
\item Create threads, assign beginning and end indices
\item For each thread
\begin{itemize}
\item Sequentially search for x
\begin{itemize}
\item If found, return index of x
\item Otherwise, return -1
\end{itemize}
\end{itemize}
\end{itemize}
\end{enumerate}
